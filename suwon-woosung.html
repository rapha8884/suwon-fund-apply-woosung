function doPost(e) {
  const d = JSON.parse(e.postData.contents);
  const s = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  s.appendRow([d.receipt_num, d.company_name, d.ceo_name, d.phone, d.email, d.industry, d.biz_age, d.fund_types, d.fund_amount, d.memo]);
  return ContentService.createTextOutput('OK');
}
